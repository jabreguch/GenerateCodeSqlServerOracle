CREATE PROCEDURE Maestros.ParametrosAdd(
@CoParametro INT OUTPUT ,
@CoEmpresa INT OUTPUT ,
@NoCampo  VARCHAR(80),
@NuOrden INT OUTPUT ,
@NoValor VARCHAR(50),
@NoNombre VARCHAR(200),
@NoSigla VARCHAR(15),
@CoUsuCre INT) AS

INSERT INTO Maestros.TrfParametros (
CoParametro,
CoEmpresa,
NoCampo,
NuOrden,
NoValor,
NoNombre,
NoSigla,
CoUsuCre)
VALUES (@CoParametro,
@CoEmpresa,
@NoCampo,
@NuOrden,
@NoValor,
@NoNombre,
@NoSigla,
@CoUsuCre) 

GO
CREATE PROCEDURE Maestros.ParametrosEdit(
@CoParametro INT OUTPUT ,
@CoEmpresa INT OUTPUT ,
@NoCampo VARCHAR(80),
@NuOrden INT OUTPUT ,
@NoValor VARCHAR(50),
@NoNombre VARCHAR(200),
@NoSigla VARCHAR(15),
@CoUsuMod INT) AS

UPDATE Maestros.TrfParametros
SET CoParametro=@CoParametro,
CoEmpresa=@CoEmpresa,
NuOrden=@NuOrden,
NoValor=@NoValor,
NoNombre=@NoNombre,
NoSigla=@NoSigla,
CoUsuMod= @CoUsuMod,FeMod= getdate()
WHERE NoCampo= @NoCampo 

GO
CREATE PROCEDURE Maestros.ParametrosDel(
@NoCampo  VARCHAR(80),
@CoUsuEli INT) AS

UPDATE Maestros.TrfParametros
SET FlSitReg='I',
CoUsuEli = @CoUsuEli,
FeEli= getdate()
WHERE NoCampo= @NoCampo 

GO
CREATE PROCEDURE Maestros.ParametrosEditEst(
@NoCampo  VARCHAR(80),
@CoEstReg INT,
@CoUsuMod INT) AS

UPDATE Maestros.TrfParametros
SET CoEstReg = @CoEstReg,
CoUsuMod= @CoUsuMod,
FeMod= getdate()
WHERE NoCampo= @NoCampo 

GO
CREATE PROCEDURE Maestros.ParametrosGets AS

SELECT NoCampo,
TrfParametros.CoParametro,
TrfParametros.CoEmpresa,
TrfParametros.NuOrden,
TrfParametros.NoValor,
TrfParametros.NoNombre,
TrfParametros.NoSigla,
TrfParametros.CoEstReg, TrfParametros.CoUsuEli, TrfParametros.CoUsuMod, TrfParametros.CoUsuCre, TrfParametros.FeCre, TrfParametros.FeEli, TrfParametros.FeMod, TrfParametros.FlSitReg
FROM  Maestros.TrfParametros
WHERE TrfParametros.FlSitReg ='A'

GO
CREATE PROCEDURE Maestros.ParametrosGet @NoCampo  VARCHAR(80) AS

SELECT NoCampo,
TrfParametros.CoParametro,
TrfParametros.CoEmpresa,
TrfParametros.NuOrden,
TrfParametros.NoValor,
TrfParametros.NoNombre,
TrfParametros.NoSigla,
TrfParametros.CoEstReg, TrfParametros.CoUsuEli, TrfParametros.CoUsuMod, TrfParametros.CoUsuCre, TrfParametros.FeCre, TrfParametros.FeEli, TrfParametros.FeMod, TrfParametros.FlSitReg
FROM  Maestros.TrfParametros
WHERE NoCampo= @NoCampo
