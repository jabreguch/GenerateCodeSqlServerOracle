CREATE PROCEDURE Condominio.CocheraAdd(
@CoCochera  INT OUTPUT ,
@CoCondominio INT OUTPUT ,
@CoEdificio INT OUTPUT ,
@NuCochera VARCHAR(10),
@NuArea INT OUTPUT ,
@NuPiso INT OUTPUT ,
@CoUsuCre INT) AS

INSERT INTO Condominio.TblCochera (
CoCochera,
CoCondominio,
CoEdificio,
NuCochera,
NuArea,
NuPiso,
CoUsuCre)
VALUES (@CoCochera,
@CoCondominio,
@CoEdificio,
@NuCochera,
@NuArea,
@NuPiso,
@CoUsuCre) 

GO
CREATE PROCEDURE Condominio.CocheraEdit(
@CoCochera INT OUTPUT ,
@CoCondominio INT OUTPUT ,
@CoEdificio INT OUTPUT ,
@NuCochera VARCHAR(10),
@NuArea INT OUTPUT ,
@NuPiso INT OUTPUT ,
@CoUsuMod INT) AS

UPDATE Condominio.TblCochera
SET CoCondominio=@CoCondominio,
CoEdificio=@CoEdificio,
NuCochera=@NuCochera,
NuArea=@NuArea,
NuPiso=@NuPiso,
CoUsuMod= @CoUsuMod,FeMod= getdate()
WHERE CoCochera= @CoCochera 

GO
CREATE PROCEDURE Condominio.CocheraDel(
@CoCochera  INT OUTPUT ,
@CoUsuEli INT) AS

UPDATE Condominio.TblCochera
SET FlSitReg='I',
CoUsuEli = @CoUsuEli,
FeEli= getdate()
WHERE CoCochera= @CoCochera 

GO
CREATE PROCEDURE Condominio.CocheraEditEst(
@CoCochera  INT OUTPUT ,
@CoEstReg INT,
@CoUsuMod INT) AS

UPDATE Condominio.TblCochera
SET CoEstReg = @CoEstReg,
CoUsuMod= @CoUsuMod,
FeMod= getdate()
WHERE CoCochera= @CoCochera 

GO
CREATE PROCEDURE Condominio.CocheraGets AS

SELECT CoCochera,
TblCochera.CoCondominio,
TblCochera.CoEdificio,
TblCochera.NuCochera,
TblCochera.NuArea,
TblCochera.NuPiso,
TblCochera.CoEstReg, TblCochera.CoUsuEli, TblCochera.CoUsuMod, TblCochera.CoUsuCre, TblCochera.FeCre, TblCochera.FeEli, TblCochera.FeMod, TblCochera.FlSitReg
FROM  Condominio.TblCochera
WHERE TblCochera.FlSitReg ='A'

GO
CREATE PROCEDURE Condominio.CocheraGet @CoCochera  INT OUTPUT  AS

SELECT CoCochera,
TblCochera.CoCondominio,
TblCochera.CoEdificio,
TblCochera.NuCochera,
TblCochera.NuArea,
TblCochera.NuPiso,
TblCochera.CoEstReg, TblCochera.CoUsuEli, TblCochera.CoUsuMod, TblCochera.CoUsuCre, TblCochera.FeCre, TblCochera.FeEli, TblCochera.FeMod, TblCochera.FlSitReg
FROM  Condominio.TblCochera
WHERE CoCochera= @CoCochera
